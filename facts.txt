;;; DEFTEMPLATES - Fact structure definitions for clarity and traceability

(deftemplate symptom
  "A symptom that can be reported by the user."
  (slot id (type SYMBOL))
  (slot description (type STRING))
  (slot category (type SYMBOL)))

(deftemplate component
  "Hardware component with its typical failure characteristics."
  (slot name (type SYMBOL))
  (slot type (type SYMBOL))
  (slot failure-mode (type STRING)))

(deftemplate failure-scenario
  "A diagnosable hardware failure scenario."
  (slot id (type SYMBOL))
  (slot name (type STRING))
  (slot module (type SYMBOL))
  (slot recommended-action (type STRING)))

(deftemplate symptom-indicator
  "D1 crisp mapping from symptom to likely scenario."
  (slot symptom-id (type SYMBOL))
  (slot indicates-scenario (type SYMBOL))
  (slot strength (type SYMBOL)))

(deftemplate cf-evidence
  "D2 probabilistic mapping from symptom evidence to scenario support."
  (slot symptom-id (type SYMBOL))
  (slot scenario-id (type SYMBOL))
  (slot weight (type NUMBER))
  (slot rationale (type STRING)))

(deftemplate fuzzy-variable
  "D2 fuzzy input variable collected during diagnosis."
  (slot name (type SYMBOL))
  (slot description (type STRING))
  (slot units (type STRING)))

(deftemplate fuzzy-set
  "Trapezoidal fuzzy set a-b-c-d used for membership calculation."
  (slot variable (type SYMBOL))
  (slot label (type SYMBOL))
  (slot a (type NUMBER))
  (slot b (type NUMBER))
  (slot c (type NUMBER))
  (slot d (type NUMBER)))

(deftemplate reported-symptom
  "Runtime user symptom with optional certainty factor."
  (slot symptom-id (type SYMBOL))
  (slot observed (type SYMBOL))
  (slot user-cf (type NUMBER) (default 1.0)))

(deftemplate sensor-reading
  "Runtime quantitative reading used by fuzzy rules."
  (slot name (type SYMBOL))
  (slot value (type NUMBER)))

(deftemplate fuzzy-assessment
  "Intermediate membership degree produced by fuzzification."
  (slot variable (type SYMBOL))
  (slot label (type SYMBOL))
  (slot degree (type NUMBER)))

(deftemplate diagnostic-conclusion
  "Unified conclusion record for D1 crisp, D2 probabilistic, and D2 fuzzy rules."
  (slot scenario-id (type SYMBOL))
  (slot theory (type SYMBOL))
  (slot score (type NUMBER))
  (slot label (type SYMBOL))
  (slot rule-id (type SYMBOL))
  (slot recommended-action (type STRING))
  (slot rationale (type STRING)))

(deftemplate session-state
  "Tracks diagnostic session lifecycle for debugging and testability."
  (slot phase (type SYMBOL))
  (slot message (type STRING)))

(deftemplate prompt-shown
  "Prevents repeated helper prompts."
  (slot type (type SYMBOL)))
