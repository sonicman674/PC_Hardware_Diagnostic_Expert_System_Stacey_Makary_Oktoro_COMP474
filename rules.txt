(deffunction clamp-score (?value)
  (if (< ?value 0.0) then
      0.0
   else
      (if (> ?value 1.0) then
          1.0
       else
          ?value)))

(deffunction strength-to-score (?strength)
  (if (eq ?strength strong) then
      0.90
   else
      (if (eq ?strength moderate) then
          0.65
       else
          0.45)))

(deffunction score-label (?score)
  (if (>= ?score 0.90) then
      very-strong
   else
      (if (>= ?score 0.75) then
          strong
       else
          (if (>= ?score 0.55) then
              moderate
           else
              weak))))

(deffunction combine-cf (?a ?b)
  (bind ?aa (clamp-score ?a))
  (bind ?bb (clamp-score ?b))
  (clamp-score (+ ?aa (* ?bb (- 1.0 ?aa)))))

(deffunction fuzzy-and (?a ?b)
  (if (< ?a ?b) then ?a else ?b))

(deffunction trapezoid-membership (?x ?a ?b ?c ?d)
  (if (or (<= ?x ?a) (>= ?x ?d)) then
      0.0
   else
      (if (and (>= ?x ?b) (<= ?x ?c)) then
          1.0
       else
          (if (< ?x ?b) then
              (/ (- ?x ?a) (- ?b ?a))
           else
              (/ (- ?d ?x) (- ?d ?c))))))

(defrule init-diagnostic-session
  "Starts a session once symptoms or sensor readings exist."
  (or
    (reported-symptom (observed yes))
    (sensor-reading))
  (not (session-state (phase ?phase)))
  =>
  (assert (session-state (phase started) (message "Diagnostic session initialized")))
  (printout t crlf "*** DIAGNOSTIC SESSION STARTED ***" crlf))

(defrule diagnose-crisp-from-symptom
  "D1 baseline rule: map a reported symptom to one crisp conclusion."
  (reported-symptom (symptom-id ?sid) (observed yes))
  (symptom-indicator (symptom-id ?sid) (indicates-scenario ?scenario) (strength ?strength))
  (failure-scenario (id ?scenario) (name ?name) (recommended-action ?action))
  (not (diagnostic-conclusion (scenario-id ?scenario) (theory crisp)))
  =>
  (bind ?score (strength-to-score ?strength))
  (assert
    (diagnostic-conclusion
      (scenario-id ?scenario)
      (theory crisp)
      (score ?score)
      (label ?strength)
      (rule-id ?sid)
      (recommended-action ?action)
      (rationale "Crisp D1 symptom-indicator matched a directly reported symptom.")))
  (printout t "D1 CRISP: " ?name " (score: " ?score ", label: " ?strength ")" crlf))

(defrule strengthen-crisp-psu-diagnosis
  "Fixes the D1 multi-symptom limitation by replacing the weaker crisp conclusion."
  (reported-symptom (symptom-id no-power-at-all) (observed yes))
  (reported-symptom (symptom-id post-beep-codes) (observed yes))
  ?old <- (diagnostic-conclusion (scenario-id psu-failure) (theory crisp) (score ?score))

(defrule fuzzify-reading
  "Generic fuzzification rule for all fuzzy variables."
  (sensor-reading (name ?variable) (value ?value))
  (fuzzy-set (variable ?variable) (label ?label) (a ?a) (b ?b) (c ?c) (d ?d))
  (not (fuzzy-assessment (variable ?variable) (label ?label)))
  (test (> (trapezoid-membership ?value ?a ?b ?c ?d) 0.0))
  =>
  (bind ?degree (trapezoid-membership ?value ?a ?b ?c ?d))
  (assert (fuzzy-assessment (variable ?variable) (label ?label) (degree ?degree)))
  (printout t "FUZZY INPUT: " ?variable " is " ?label " with degree " ?degree crlf))

;;; SUMMARY, VERIFICATION, AND HELPER RULES

(defrule print-diagnostic-summary
  "Prints a traceable summary for each conclusion."
  (declare (salience -10))
  (diagnostic-conclusion
    (scenario-id ?scenario)
    (theory ?theory)
    (score ?score)
    (label ?label)
    (rule-id ?rule)
    (recommended-action ?action)
    (rationale ?rationale))
  =>
  (printout t "  -> [" ?theory "/" ?rule "] " ?scenario
            " score=" ?score " label=" ?label crlf)
  (printout t "     Action: " ?action crlf)
  (printout t "     Why: " ?rationale crlf))

(defrule suggest-next-steps
  "Suggest a verification step whenever the inferred score is high."
  (declare (salience -15))
  (diagnostic-conclusion (score ?score))
  (test (>= ?score 0.75))
  =>
  (printout t "  [Verification] Consider a known-good swap test or independent measurement for this high-scoring diagnosis." crlf))

(defrule session-complete
  "Mark the session complete once at least one diagnosis has been produced."
  (declare (salience -20))
  (diagnostic-conclusion)
  ?session <- (session-state (phase started) (message ?msg))
  =>
  (retract ?session)
  (assert (session-state (phase complete) (message "Diagnosis complete")))
  (printout t crlf "*** DIAGNOSTIC SESSION COMPLETE ***" crlf))

(defrule no-evidence-reported
  "Prompt the user only when neither symptoms nor fuzzy sensor readings exist."
  (not (reported-symptom (observed yes)))
  (not (sensor-reading))
  (not (prompt-shown (type no-evidence)))
  =>
  (assert (prompt-shown (type no-evidence)))
  (printout t "No evidence reported yet." crlf)
  (printout t "Assert symptoms with certainty factors, for example:" crlf)
  (printout t "  (assert (reported-symptom (symptom-id no-power-at-all) (observed yes) (user-cf 0.9)))" crlf)
  (printout t "Assert fuzzy readings, for example:" crlf)
  (printout t "  (assert (sensor-reading (name cpu-temperature) (value 95)))" crlf)
  (printout t "Valid symptom ids are the same 15 ids shown by run-diagnosis." crlf))
